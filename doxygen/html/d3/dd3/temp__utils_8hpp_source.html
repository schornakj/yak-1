123 __kf_device__ void computeRoots3(float c0, float c1, float c2, float3& roots)
124 {
125  if (fabsf(c0) < numeric_limits<float>::epsilon()) // one root is 0 -> quadratic equation
126  {
127  computeRoots2(c2, c1, roots);
128  }
129  else
130  {
131  const float s_inv3 = 1.f / 3.f;
132  const float s_sqrt3 = sqrtf(3.f);
133  // Construct the parameters used in classifying the roots of the equation
134  // and in solving the equation for the roots in closed form.
135  float c2_over_3 = c2 * s_inv3;
136  float a_over_3 = (c1 - c2 * c2_over_3) * s_inv3;
137  if (a_over_3 > 0.f)
138  a_over_3 = 0.f;
139 
140  float half_b = 0.5f * (c0 + c2_over_3 * (2.f * c2_over_3 * c2_over_3 - c1));
141 
142  float q = half_b * half_b + a_over_3 * a_over_3 * a_over_3;
143  if (q > 0.f)
144  q = 0.f;
145 
146  // Compute the eigenvalues by solving for the roots of the polynomial.
147  float rho = sqrtf(-a_over_3);
148  float theta = atan2f(sqrtf(-q), half_b) * s_inv3;
149  float cos_theta = __cosf(theta);
150  float sin_theta = __sinf(theta);
151  roots.x = c2_over_3 + 2.f * rho * cos_theta;
152  roots.y = c2_over_3 - rho * (cos_theta + s_sqrt3 * sin_theta);
153  roots.z = c2_over_3 - rho * (cos_theta - s_sqrt3 * sin_theta);
154 
155  // Sort in increasing order.
156  if (roots.x >= roots.y)
157  swap(roots.x, roots.y);
158 
159  if (roots.y >= roots.z)
160  {
161  swap(roots.y, roots.z);
162 
163  if (roots.x >= roots.y)
164  swap(roots.x, roots.y);
165  }
166  if (roots.x <= 0) // eigenval for symetric positive semi-definite matrix can not be negative! Set it to 0
167  computeRoots2(c2, c1, roots);
168  }
